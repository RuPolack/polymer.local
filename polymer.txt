-- Создаем функцию для шифрования имени
CREATE OR REPLACE FUNCTION encrypt_name(user_name TEXT)
RETURNS TEXT AS $$
DECLARE
    encrypted_bytes BYTEA;
BEGIN
    -- Шифруем с использованием пароля 'polymer'
    SELECT pgp_sym_encrypt(user_name, 'polymer') INTO encrypted_bytes;
    -- Конвертируем в base64 для хранения как текст
    RETURN encode(encrypted_bytes, 'base64');
END;
$$ LANGUAGE plpgsql;

-- Создаем функцию для расшифровки имени
CREATE OR REPLACE FUNCTION decrypt_name(encrypted_text TEXT)
RETURNS TEXT AS $$
DECLARE
    encrypted_bytes BYTEA;
BEGIN
    -- Декодируем из base64
    SELECT decode(encrypted_text, 'base64') INTO encrypted_bytes;
    -- Расшифровываем с использованием пароля 'polymer'
    RETURN pgp_sym_decrypt(encrypted_bytes, 'polymer');
EXCEPTION
    WHEN others THEN
        RETURN 'Ошибка дешифровки';
END;
$$ LANGUAGE plpgsql;

-- Создаем представление для просмотра расшифрованных данных
CREATE OR REPLACE VIEW decrypted_results AS
SELECT 
    r.id,
    decrypt_name(r.encrypted_name) as user_name,
    r.profession_id,
    r.question_id,
    r.user_answer,
    r.answer_date,
    p.name as profession_name,
    q.question_text
FROM results r
LEFT JOIN professions p ON r.profession_id = p.id
LEFT JOIN questions q ON r.question_id = q.id;

-- Создаем индексы для оптимизации запросов
CREATE INDEX idx_results_encrypted ON results(encrypted_name);
CREATE INDEX idx_results_date ON results(answer_date);
CREATE INDEX idx_results_profession ON results(profession_id);
CREATE INDEX idx_results_question ON results(question_id);
